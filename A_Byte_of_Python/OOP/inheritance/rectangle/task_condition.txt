 Клас Point2 описано наступним чином:
class Point2:
 """Клас реалізує точку площини"""
 def __init__(self, x, y):
 self._x = x
 self._y = y
 def get_x(self):
 """Повернути координату х"""
 return self._x
 def get_y(self):
 """Повернути координату y"""
 return self._y
 def __str__(self):
    """Повернути рядок представлення точки"""
    return "({}, {})".format(self._x, self._y)
